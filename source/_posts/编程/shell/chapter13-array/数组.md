### 定义及打印

```shell
#!/bin/bash
# 基础定义
array=(one two three)
# 动态数组
array2=($(ls ../))

# 繁琐赋值
array3=([1]=one [2]=two [3]=three)

# 分别赋值
array4[1]=one; array4[2]=two; array4[3]=three;

# 打印数组
echo ${array[*]}
echo ${array[@]}
echo ${array2[@]}
echo ${array3[@]}
echo ${array4[@]}
```

### 数组增删改查

```shell
#!/bin/bash
array=(one two three)
# 增
array[3]=test
echo ${array[*]}
# 删
unset array[2]
echo ${array[*]}
# 改
array[1]=change
echo ${array[*]}
# 查
echo ${array[1]}
```

### 数组常用api

```shell
#!/bin/bash
array=(one two three)

# 数组长度
echo ${#array}

# 截取数组
echo ${array[*]}
# 截取位置到结尾
echo ${array[*]:1}
# 截取位置及长度
echo ${array[*]:1:1}

# 删除数组
array[4]=one4one
# 从头开始 one change test one4one
echo ${array[*]} 
echo ${array[*]#one} #change test 4one
echo ${array[*]##one*} # change test
# 从尾开始 one change test one4one
echo ${array[*]}
echo ${array[*]%one} # change test one4
echo ${array[*]%%one*} # change test

# 替换数组
echo ${array[*]}
# 仅第一次匹配到替换
echo ${array[*]/one/one1}
# 匹配到所有替换
echo ${array[*]//one/one1}
```
